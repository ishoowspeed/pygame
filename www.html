import pygame
import random

# Inicjalizacja Pygame
pygame.init()

# Ustawienia okna
WIDTH, HEIGHT = 800, 600
win = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Mini RPG")

# Kolory
WHITE = (255, 255, 255)
BLACK = (0, 0, 0)
RED = (200, 0, 0)
GREEN = (0, 200, 0)
BLUE = (50, 50, 255)

# Gracz
player_health = 100
player_max_health = 100
player_damage = 20
player_x = 100
player_y = 250

# Przeciwnik
enemy_health = 50
enemy_max_health = 50
enemy_damage = 10
enemy_x = 550
enemy_y = 250

# Czcionki
font = pygame.font.SysFont('Arial', 24)

# Funkcja rysująca postacie i HUD
def draw_player():
    pygame.draw.rect(win, BLUE, (player_x, player_y, 60, 100))
    health_text = font.render(f"Gracz: {player_health}", True, WHITE)
    win.blit(health_text, (player_x, player_y - 30))

def draw_enemy():
    pygame.draw.rect(win, RED, (enemy_x, enemy_y, 60, 100))
    health_text = font.render(f"Przeciwnik: {enemy_health}", True, WHITE)
    win.blit(health_text, (enemy_x, enemy_y - 30))

def draw_background():
    win.fill(BLACK)

# Funkcja ataku
def attack(target):
    global enemy_health, player_health
    damage = random.randint(1, player_damage)
    if target == "enemy":
        enemy_health -= damage
        if enemy_health < 0:
            enemy_health = 0
    elif target == "player":
        player_health -= random.randint(1, enemy_damage)
        if player_health < 0:
            player_health = 0

# Główna pętla gry
def main():
    global player_health, enemy_health
    clock = pygame.time.Clock()
    run = True
    turn = "player"

    while run:
        clock.tick(60)
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                run = False

        keys = pygame.key.get_pressed()
        if keys[pygame.K_SPACE]:
            if turn == "player" and enemy_health > 0:
                attack("enemy")
                turn = "enemy"
            elif turn == "enemy" and player_health > 0:
                attack("player")
                turn = "player"

        draw_background()
        draw_player()
        draw_enemy()

        if player_health <= 0:
            game_over_text = font.render("Przegrałeś!", True, RED)
            win.blit(game_over_text, (WIDTH // 2 - 70, HEIGHT // 2))
        elif enemy_health <= 0:
            game_over_text = font.render("Wygrałeś!", True, GREEN)
            win.blit(game_over_text, (WIDTH // 2 - 70, HEIGHT // 2))
        else:
            turn_text = font.render(f"Tura: {turn.capitalize()}", True, WHITE)
            win.blit(turn_text, (WIDTH // 2 - 60, 20))

        pygame.display.update()

    pygame.quit()

if __name__ == "__main__":
    main()
